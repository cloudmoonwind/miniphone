# 沉浸式角色互动系统 - 项目文档

## 1. 项目概述

### 1.1 项目信息
- **项目名称**：Immersive Character System (ICS)
- **版本**：v0.1.0-alpha
- **开发语言**：TypeScript + JavaScript
- **许可协议**：MIT
- **目标平台**：Web + Desktop (Electron)

### 1.2 项目目标
构建一个超越传统聊天界面的沉浸式角色互动系统，通过多维度的功能模块和智能的AI驱动，让虚拟角色真正"活"起来。

## 4. 数据模型设计

### 4.1 设计说明

本项目采用**存储无关**的数据模型设计理念：
- **数据模型**：定义了数据的结构和关系（下方Schema）
- **存储实现**：可以是文件存储、数据库、或其他方式

**当前实现**：文件存储（JSON格式）
**未来扩展**：PostgreSQL + Prisma（详见第7章）

**核心原则**：
- 无论用什么存储方式，数据结构保持一致
- 下方的Schema既是数据库设计，也是JSON文件的结构模板
- 切换存储方式时，只需改变数据访问层实现，业务逻辑不变

### 4.2 数据模型Schema

以下Schema使用Prisma语法编写，既可作为数据库迁移文件，也可作为JSON文件的结构参考：

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ========== 用户相关 ==========

model User {
  id        Int       @id @default(autoincrement())
  username  String    @unique
  email     String?   @unique
  password  String    // bcrypt哈希
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt
  settings  Json?     // 用户设置

  characters Character[]
  personas   Persona[]
}

// ========== 角色相关 ==========

model Character {
  id            Int       @id @default(autoincrement())
  userId        Int
  name          String
  avatarUrl     String?
  personality   String    @db.Text
  background    String    @db.Text
  systemPrompt  String    @db.Text
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  user          User      @relation(fields: [userId], references: [id])
  
  conversations  Conversation[]
  memories       Memory[]
  attributes     CharacterAttribute[]
  lifeLogs       CharacterLifeLog[]
  schedules      Schedule[]
  systemTasks    CharacterSystemTask[]
  personas       CharacterPersona[]
  emotions       CharacterEmotion[]
  socialPosts    SocialPost[]
  npcInteractions NPCInteraction[]
  dreams         Dream[]
  items          Item[]
  room           Room?
  
  @@index([userId])
}

model Persona {
  id              Int       @id @default(autoincrement())
  userId          Int
  name            String
  avatarUrl       String?
  personalityTags Json      // ["温柔", "幽默"]
  background      String?   @db.Text
  createdAt       DateTime  @default(now())

  user            User      @relation(fields: [userId], references: [id])
  
  conversations   Conversation[]
  memories        Memory[]
  attributes      CharacterAttribute[]
  
  @@index([userId])
}

// ========== 角色马甲（新增）==========

model CharacterPersona {
  id              Int       @id @default(autoincrement())
  characterId     Int
  name            String
  type            String    // 'work' | 'private' | 'hobby'
  avatarUrl       String?
  description     String?   @db.Text
  visibility      String    @default("hidden")  // 'public' | 'hidden' | 'discoverable'
  createdAt       DateTime  @default(now())

  character       Character @relation(fields: [characterId], references: [id])
  
  @@index([characterId])
}

// ========== 对话相关 ==========

model Conversation {
  id            Int       @id @default(autoincrement())
  characterId   Int
  personaId     Int?      // NULL表示用户真实身份
  mode          String    // 'online' | 'offline'
  startedAt     DateTime  @default(now())
  lastMessageAt DateTime  @updatedAt

  character     Character  @relation(fields: [characterId], references: [id])
  persona       Persona?   @relation(fields: [personaId], references: [id])
  
  messages      Message[]
  
  @@index([characterId, personaId])
}

model Message {
  id              Int       @id @default(autoincrement())
  conversationId  Int
  sender          String    // 'user' | 'character'
  content         String    @db.Text
  mode            String    // 'online' | 'offline'
  timestamp       DateTime  @default(now())
  metadata        Json?     // 特殊标记、附件等

  conversation    Conversation @relation(fields: [conversationId], references: [id])
  
  @@index([conversationId, timestamp])
}

// ========== 记忆系统 ==========

model Memory {
  id          Int       @id @default(autoincrement())
  characterId Int
  personaId   Int?      // 记忆归属的马甲
  category    String    // 'short_term' | 'episodic' | 'semantic' | 'relationship'
  content     String    @db.Text
  importance  Int       // 1-10
  timestamp   DateTime  @default(now())
  tags        Json?     // 关键词标签
  embedding   String?   @db.Text  // 向量嵌入（JSON字符串）

  character   Character @relation(fields: [characterId], references: [id])
  persona     Persona?  @relation(fields: [personaId], references: [id])
  
  @@index([characterId, personaId, category])
  @@index([timestamp])
}

// ========== 数值系统 ==========

model CharacterAttribute {
  id          Int       @id @default(autoincrement())
  characterId Int
  personaId   Int?
  affection   Int       @default(0)  // 好感度
  trust       Int       @default(0)  // 信任度
  intimacy    Int       @default(0)  // 亲密度
  mood        Int       @default(0)  // 情绪值 (-50~50)
  dependency  Int       @default(0)  // 依赖度（隐藏）
  boredom     Int       @default(0)  // 厌倦值（隐藏）
  mystery     Int       @default(50) // 神秘感（隐藏）
  updatedAt   DateTime  @updatedAt

  character   Character @relation(fields: [characterId], references: [id])
  persona     Persona?  @relation(fields: [personaId], references: [id])
  
  @@unique([characterId, personaId])
}

// ========== 角色生活日志 ==========

model CharacterLifeLog {
  id           Int       @id @default(autoincrement())
  characterId  Int
  timestamp    DateTime  @default(now())
  activityType String    // 'daily' | 'social' | 'emotion' | 'user_related'
  content      String    @db.Text
  location     String?
  emotion      String?
  visibility   Boolean   @default(true)  // 是否对用户可见

  character    Character @relation(fields: [characterId], references: [id])
  
  @@index([characterId, timestamp])
}

// ========== 情绪系统（新增）==========

model CharacterEmotion {
  id          Int       @id @default(autoincrement())
  characterId Int
  timestamp   DateTime  @default(now())
  valence     Int       // 愉悦度 -100~100
  arousal     Int       // 激活度 0~100
  dominance   Int       // 控制度 -100~100
  label       String?   // 复合情绪标签："快乐"、"悲伤"等
  trigger     String?   @db.Text  // 触发原因
  
  character   Character @relation(fields: [characterId], references: [id])
  
  @@index([characterId, timestamp])
}

// ========== 社交网络（新增）==========

model SocialPost {
  id          Int       @id @default(autoincrement())
  characterId Int
  content     String    @db.Text
  type        String    // 'moment' | 'status' | 'share'
  mediaUrl    String?   
  visibility  String    @default("public")  // 'public' | 'friends'
  createdAt   DateTime  @default(now())
  
  character   Character @relation(fields: [characterId], references: [id])
  likes       PostLike[]
  comments    PostComment[]
  
  @@index([characterId, createdAt])
}

model PostLike {
  id        Int       @id @default(autoincrement())
  postId    Int
  userId    Int
  personaId Int?      // 哪个马甲点的赞
  createdAt DateTime  @default(now())

  @@unique([postId, userId, personaId])
}

model PostComment {
  id          Int       @id @default(autoincrement())
  postId      Int
  authorType  String    // 'user' | 'character' | 'npc'
  authorId    Int
  content     String    @db.Text
  createdAt   DateTime  @default(now())
  
  post        SocialPost @relation(fields: [postId], references: [id])
  
  @@index([postId])
}

model NPCInteraction {
  id          Int       @id @default(autoincrement())
  characterId Int
  npcId       Int
  type        String    // 'chat' | 'call' | 'meetup'
  summary     String    @db.Text
  timestamp   DateTime  @default(now())
  
  character   Character @relation(fields: [characterId], references: [id])
  npc         NPC       @relation(fields: [npcId], references: [id])
  
  @@index([characterId, timestamp])
}

model NPC {
  id            Int       @id @default(autoincrement())
  name          String
  relationship  String    // 与主角色的关系
  personality   String    @db.Text
  avatarUrl     String?
  
  interactions  NPCInteraction[]
  
  @@index([name])
}

// ========== 关系网络（新增）==========

model CharacterRelationship {
  id              Int       @id @default(autoincrement())
  characterId1    Int
  characterId2    Int

  intimacy        Int       @default(50)  // 亲密度 0-100
  status          String    @default("neutral")  // 'good' | 'neutral' | 'bad'
  description     String?   @db.Text
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt

  @@index([characterId1])
  @@index([characterId2])

// ========== 时间事件（新增）==========

model TimeEvent {
  id          Int       @id @default(autoincrement())
  type        String    // 'birthday' | 'anniversary' | 'holiday'
  name        String
  date        DateTime  @db.Date
  isAnnual    Boolean   @default(true)  // 是否每年重复

model Weather {
  id          Int       @id @default(autoincrement())
  date        DateTime  @db.Date
  condition   String    // 'sunny' | 'rainy' | 'cloudy' | 'snowy'
```

## 6. 核心服务实现

### 6.1 AI服务 (server/src/services/ai/)

#### 6.1.1 Prompt构建器

```typescript
// server/src/services/ai/promptBuilder.ts

import { Memory, CharacterAttribute } from '@prisma/client';

interface PromptContext {
  character: {
    name: string;
    personality: string;
    background: string;
    systemPrompt: string;
  };
  conversation: {
    mode: 'online' | 'offline';
    personaName?: string;
  };
  memories: Memory[];
  attributes: CharacterAttribute;
  currentTime: Date;
}

export class PromptBuilder {
  buildSystemPrompt(context: PromptContext): string {
    const { character, conversation, memories, attributes, currentTime } = context;
    
    let prompt = `# 角色设定
你是${character.name}。

## 性格
${character.personality}

## 背景
${character.background}

## 核心指令
${character.systemPrompt}

# 当前情境
`;

    // 添加模式说明
    if (conversation.mode === 'online') {
      prompt += `
## 对话模式：线上（短信/社交软件）
- 保持简短（20-150字）
- 使用口语化表达
- 可以使用表情符号
- 旁白极少，仅在必要时用 [旁白: ...] 标记
`;
    } else {
      prompt += `
## 对话模式：线下（面对面）
- 详细描写场景、动作、心理
- 长度200-800字
- 使用第三人称叙事
- 可穿插线上内容，用 [手机消息: ...] 标记
`;
    }

    // 添加对话对象信息
    if (conversation.personaName) {
      prompt += `
## 对话对象
你正在和 ${conversation.personaName} 对话。这是你们${this.getRelationshipPhrase(attributes.affection)}的关系。
`;
    }

    // 添加当前状态
    prompt += `
# 你的当前状态
- 情绪：${this.getMoodDescription(attributes.mood)}
- 对TA的好感度：${attributes.affection}/100
- 信任度：${attributes.trust}/100
- 时间：${currentTime.toLocaleString('zh-CN')}
`;

    // 添加相关记忆
    if (memories.length > 0) {
      prompt += `
# 相关记忆
`;
      memories.forEach((mem, idx) => {
        prompt += `${idx + 1}. ${mem.content}\n`;
      });
    }

    prompt += `
# 回复要求
- 保持角色一致性
- 根据当前情绪调整语气
- 自然地融入记忆中的信息
- 不要生成用户的对话或行动
`;

    return prompt;
  }

  private getRelationshipPhrase(affection: number): string {
    if (affection < 20) return '刚认识';
    if (affection < 40) return '初步熟悉';
    if (affection < 60) return '普通朋友';
    if (affection < 80) return '亲密朋友';
    return '非常亲密';
  }

  private getMoodDescription(mood: number): string {
    if (mood < -30) return '很糟糕';
    if (mood < -10) return '有点低落';
    if (mood < 10) return '平静';
    if (mood < 30) return '愉快';
    return '非常开心';
  }
}
```

#### 6.1.2 AI Provider抽象

```typescript

export interface AIMessage {
  role: 'system' | 'user' | 'assistant';

export interface AIResponse {

export abstract class AIProvider {
```

```typescript
// server/src/services/ai/providers/openai.ts

import OpenAI from 'openai';
import { AIProvider, AIMessage, AIResponse } from './base';

export class OpenAIProvider extends AIProvider {
  private client: OpenAI;
  
  constructor(apiKey: string) {
    super();
    this.client = new OpenAI({ apiKey });
  }
  
  async chat(messages: AIMessage[], stream = false): Promise<AIResponse | AsyncIterator<string>> {
    const response = await this.client.chat.completions.create({
      model: 'gpt-4-turbo-preview',
      messages,
      stream,
    });
    
    if (stream) {
      return this.streamResponse(response);
    } else {
      const result = response as OpenAI.Chat.Completions.ChatCompletion;
      return {
        content: result.choices[0].message.content || '',
        usage: {
          promptTokens: result.usage?.prompt_tokens || 0,
          completionTokens: result.usage?.completion_tokens || 0,
          totalTokens: result.usage?.total_tokens || 0,
        },
      };
    }
  }
  
  private async *streamResponse(stream: any): AsyncIterator<string> {
    for await (const chunk of stream) {
      const delta = chunk.choices[0]?.delta?.content;
      if (delta) {
        yield delta;
      }
    }
  }
  
  async embedText(text: string): Promise<number[]> {
    const response = await this.client.embeddings.create({
```

```typescript
    yesterday.setHours(0, 0, 0, 0);

          lt: today,

        tags: fact.tags,
```

```typescript
      this.prisma.characterAttribute.findFirst({

      // 高重要性记忆
      this.prisma.memory.findMany({
        where: {

        },
        orderBy: { timestamp: 'desc' },
        take: 5,

      memories,
      mainTrigger: this.identifyMainTrigger(emotions, events),

最近情绪：

生成要求：

  ],

  "hiddenMessage": "..."
}`;
  }
  
  private parseDreamContent(content: string): DreamContent {
    try {
      // 尝试解析JSON
      const cleaned = content.replace(/```json|```/g, '').trim();
      return JSON.parse(cleaned);
    } catch (error) {
      // 如果解析失败，返回默认结构
      return {
        scenes: [{
          order: 1,
          description: content,
          emotion: '未知',
          symbols: [],
        }],
        overallMood: '模糊',
        hiddenMessage: '需要进一步解读',
      };
    }
  }
  
  private classifyDream(content: DreamContent, data: any): { type: string; category: string } {
    // 简化分类逻辑
    const avgValence = data.emotions.reduce((sum: number, e: any) => sum + e.valence, 0) / data.emotions.length;
    
    let type = 'emotion';
    let category = 'neutral';
    
    if (avgValence < -30) {
      category = 'anxiety';
    } else if (avgValence > 30) {
      category = 'happiness';
    }
    
    // 检查是否包含用户相关内容
    if (content.scenes.some(s => s.description.includes('你') || s.description.includes('用户'))) {
      type = 'desire';
    }
    
    return { type, category };
  }

    const now = new Date();
    const dateStr = now.toISOString().split('T')[0].replace(/-/g, '');
    const random = Math.floor(Math.random() * 1000).toString().padStart(3, '0');
    return `D${dateStr}_${random}`;
  }
  
  private calculateImportance(content: DreamContent, data: any): number {
    // 基于情绪强度和内容深度计算重要性
    let importance = 5;
    
    // 情绪越极端越重要
    const avgValence = Math.abs(data.emotions.reduce((sum: number, e: any) => sum + e.valence, 0) / data.emotions.length);
    importance += Math.floor(avgValence / 20);
    
    // 包含用户相关内容加分
    if (content.hiddenMessage.includes('用户')) {
      importance += 2;
    }
    
    return Math.min(importance, 10);
  }
  
  private identifyMainTrigger(emotions: any[], events: any[]): string {
    // 找出最强烈的情绪或事件作为触发器
    if (emotions.length > 0) {
      const strongest = emotions.reduce((max, e) => 
        Math.abs(e.valence) > Math.abs(max.valence) ? e : max
      );
      return strongest.trigger || '情绪波动';
    }
    
    if (events.length > 0) {
      return events[0].content;
    }
    
    return '日常生活';
  }

          data: {
            dreamId,
            symbol: symbol.trim(),
            meaning: meaning.trim(),
          },
        });
      }
    }
  }
```

```typescript
请生成你的即时回复（20-50字），表达惊喜和感激：`;
    
    const messages = [
      { role: 'user' as const, content: prompt },
    ];
    
    const response = await this.aiProvider.chat(messages);
    const reply = typeof response === 'object' ? response.content : '';
    
    return {
      immediateReply: reply,
      emotion: '惊喜',
      affectionChange: Math.floor(Math.random() * 5) + 6,  // 6-10
    };
  }
  
  private async updateAffection(characterId: number, change: number): Promise<void> {
    await this.prisma.characterAttribute.updateMany({
      where: { characterId },
      data: {
        affection: { increment: change },
      },
    });
  }
  
  private async scheduleItemDiary(characterId: number, itemId: string): Promise<void> {
    // 创建一个生活日志，标记今晚将写关于这个物品的日记
    await this.prisma.characterLifeLog.create({
      data: {
        characterId,
        activityType: 'user_related',
        content: `收到了用户送的礼物，今晚要写进日记`,
        emotion: '开心',
        visibility: false,  // 不对用户可见
      },
    });
  }
  
  private async updateRoomPlacement(
    characterId: number,
    itemId: string,
    location: string
  ): Promise<void> {
    // 获取或创建房间
    let room = await this.prisma.room.findUnique({
      where: { characterId },
    });
    
    if (!room) {
      room = await this.prisma.room.create({
        data: {
          characterId,
          layout: { zones: ['desk', 'shelf', 'bed', 'closet'] },
        },
      });
    }
    
    // 确定摆放位置
    const zone = this.determineZone(location);
    const position = this.generatePosition(zone);
    
    // 更新或创建摆放记录
    await this.prisma.roomPlacement.upsert({
      where: {
        roomId_itemId: {
          roomId: room.id,
          itemId,
        },
      },
      update: { position },
      create: {
        roomId: room.id,
        itemId,
        position,
      },
    });
  }
  
  private determineZone(location: string): string {
    if (location.includes('桌')) return 'desk';
    if (location.includes('架')) return 'shelf';
    if (location.includes('床')) return 'bed';
    if (location.includes('柜')) return 'closet';
    return 'desk';
  }
  
  private generatePosition(zone: string): any {
    // 简化：随机生成位置
    return {
      x: Math.floor(Math.random() * 100),
      y: Math.floor(Math.random() * 100),
      zone,
    };
  }
```

## 7. 数据访问层设计

### 7.1 架构概述

#### 7.1.1 设计理念

**核心问题**：如何让应用既能用文件存储（开发阶段），又能无缝切换到数据库（生产阶段）？

**解决方案**：Repository模式 + 依赖注入

```
┌─────────────────────────────────────────┐
│         业务逻辑层 (Services)            │
│   ChatService, MemoryManager, etc.      │
└─────────────────┬───────────────────────┘
                  │ 只依赖接口
                  ↓
┌─────────────────────────────────────────┐
│       Repository接口层 (Interfaces)      │
│   IConversationRepo, ICharacterRepo     │
└─────────────────┬───────────────────────┘
                  │
        ┌─────────┴─────────┐
        ↓                   ↓
┌───────────────┐   ┌───────────────┐
│  文件存储实现  │   │  数据库实现   │
│ FileXxxRepo   │   │ DbXxxRepo     │
└───────────────┘   └───────────────┘
```

**关键点**：
- 业务逻辑**永远不知道**数据存在文件还是数据库
- 切换存储方式**只需改一行配置**
- 两种实现遵循**完全相同的接口**

```typescript
// repositories/file/FileConversationRepository.ts

import * as fs from 'fs/promises';
import * as path from 'path';
import { IConversationRepository, Conversation, Message } from '../interfaces/IConversationRepository';

export class FileConversationRepository implements IConversationRepository {
  private dataDir: string;
  
  constructor(baseDir: string = './data') {
    this.dataDir = path.join(baseDir, 'conversations');
    this.ensureDir();
  }
  
  private async ensureDir(): Promise<void> {
    await fs.mkdir(this.dataDir, { recursive: true });
  }
  
  private getFilePath(id: number): string {
    return path.join(this.dataDir, `conv_${id}.json`);
  }
  
  async findById(id: number): Promise<Conversation | null> {
    try {
      const filePath = this.getFilePath(id);
      const data = await fs.readFile(filePath, 'utf-8');
      return JSON.parse(data);
    } catch (error) {
      if ((error as any).code === 'ENOENT') {
        return null;  // 文件不存在
      }
      throw error;
    }
  }
  
  async findByCharacter(characterId: number): Promise<Conversation[]> {
    const files = await fs.readdir(this.dataDir);
    const conversations: Conversation[] = [];
    
    for (const file of files) {
      if (file.endsWith('.json')) {
        const data = await fs.readFile(path.join(this.dataDir, file), 'utf-8');
        const conv = JSON.parse(data);
        if (conv.characterId === characterId) {
          conversations.push(conv);
        }
      }
    }
    
    return conversations;
  }
  
  async findAll(): Promise<Conversation[]> {
    const files = await fs.readdir(this.dataDir);
    const conversations: Conversation[] = [];
    
    for (const file of files) {
      if (file.endsWith('.json')) {
        const data = await fs.readFile(path.join(this.dataDir, file), 'utf-8');
        conversations.push(JSON.parse(data));
      }
    }
    
    return conversations;
  }
  
  async create(data: Omit<Conversation, 'id'>): Promise<Conversation> {
    const id = await this.getNextId();
    const conversation: Conversation = {
      id,
      ...data,
      startedAt: new Date(),
      lastMessageAt: new Date(),
    };
    
    await this.save(conversation);
    return conversation;
  }
  
  async update(id: number, data: Partial<Conversation>): Promise<Conversation> {
    const existing = await this.findById(id);
    if (!existing) {
      throw new Error(`Conversation ${id} not found`);
    }
    
    const updated = { ...existing, ...data };
    await this.save(updated);
    return updated;
  }
  
  async addMessage(conversationId: number, message: Omit<Message, 'id'>): Promise<Message> {
    const conversation = await this.findById(conversationId);
    if (!conversation) {
      throw new Error(`Conversation ${conversationId} not found`);
    }
    
    const messageId = conversation.messages.length + 1;
    const newMessage: Message = {
      id: messageId,
      ...message,
      timestamp: new Date(),
    };
    
    conversation.messages.push(newMessage);
    conversation.lastMessageAt = newMessage.timestamp;
    
    await this.save(conversation);
    return newMessage;
  }
  
  async switchMode(conversationId: number, mode: 'online' | 'offline'): Promise<void> {
    await this.update(conversationId, { mode });
  }
  
  async delete(id: number): Promise<void> {
    const filePath = this.getFilePath(id);
    await fs.unlink(filePath);
  }
  
  async count(): Promise<number> {
    const files = await fs.readdir(this.dataDir);
    return files.filter(f => f.endsWith('.json')).length;
  }
  
  private async save(conversation: Conversation): Promise<void> {
    const filePath = this.getFilePath(conversation.id);
    await fs.writeFile(filePath, JSON.stringify(conversation, null, 2), 'utf-8');
  }
  
  private async getNextId(): Promise<number> {
    const counterFile = path.join('./data/metadata', 'counters.json');
    let counters: any = {};
    
    try {
      const data = await fs.readFile(counterFile, 'utf-8');
      counters = JSON.parse(data);
    } catch {
      // 文件不存在，初始化
      await fs.mkdir(path.dirname(counterFile), { recursive: true });
    }
    
    counters.conversation = (counters.conversation || 0) + 1;
    await fs.writeFile(counterFile, JSON.stringify(counters, null, 2));
    
    return counters.conversation;
  }
}
```

```typescript
import * as fs from 'fs/promises';
import * as path from 'path';

export class FileHelper {
  /**
   * 确保目录存在
   */
  static async ensureDir(dir: string): Promise<void> {
    await fs.mkdir(dir, { recursive: true });
  }
  
  /**
   * 读取JSON文件
   */
  static async readJson<T>(filePath: string): Promise<T | null> {
    try {
      const data = await fs.readFile(filePath, 'utf-8');
      return JSON.parse(data);
    } catch (error) {
      if ((error as any).code === 'ENOENT') return null;
      throw error;
    }
  }
  
  /**
   * 写入JSON文件
   */
  static async writeJson(filePath: string, data: any): Promise<void> {
    await fs.writeFile(filePath, JSON.stringify(data, null, 2), 'utf-8');
  }
  
  /**
   * 获取下一个ID
   */
  static async getNextId(counterName: string, baseDir: string = './data'): Promise<number> {
    const counterFile = path.join(baseDir, 'metadata', 'counters.json');
    await this.ensureDir(path.dirname(counterFile));
    
    let counters: any = await this.readJson(counterFile) || {};
    counters[counterName] = (counters[counterName] || 0) + 1;
    
    await this.writeJson(counterFile, counters);
    return counters[counterName];
  }
  
  /**
   * 列出目录下所有JSON文件
   */
  static async listJsonFiles(dir: string): Promise<string[]> {
    try {
      const files = await fs.readdir(dir);
      return files.filter(f => f.endsWith('.json'));
    } catch {
      return [];
    }
  }
}
```

```typescript
// scripts/migrate-to-database.ts

import RepositoryFactory from '../repositories';
import { PrismaClient } from '@prisma/client';
import * as fs from 'fs/promises';
import * as path from 'path';

async function migrateToDatabase() {
  console.log('开始迁移数据到数据库...');
  
  const prisma = new PrismaClient();
  const dataDir = './data';
  
  try {
    // 1. 迁移角色
    console.log('迁移角色数据...');
    const characterFiles = await fs.readdir(path.join(dataDir, 'characters'));
    for (const file of characterFiles) {
      const data = await fs.readFile(path.join(dataDir, 'characters', file), 'utf-8');
      const character = JSON.parse(data);
      await prisma.character.create({ data: character });
    }
    
    // 2. 迁移对话
    console.log('迁移对话数据...');
    const convFiles = await fs.readdir(path.join(dataDir, 'conversations'));
    for (const file of convFiles) {
      const data = await fs.readFile(path.join(dataDir, 'conversations', file), 'utf-8');
      const conversation = JSON.parse(data);
      
      await prisma.conversation.create({
        data: {
          ...conversation,
          messages: {
            create: conversation.messages,
          },
        },
      });
    }
    
    // 3. 迁移记忆
    console.log('迁移记忆数据...');
    // ... 类似逻辑
    
    // 4. 迁移其他数据
    // ...
    
    console.log('✅ 迁移完成！');
    console.log('请修改 .env: STORAGE_TYPE=database');
    
  } catch (error) {
    console.error('❌ 迁移失败:', error);
  } finally {
    await prisma.$disconnect();
  }
}

migrateToDatabase();
```

### 8.1 状态管理 (Zustand)

```typescript
// client/src/stores/chatStore.ts

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface Message {
  id: number;
  sender: 'user' | 'character';
  content: string;
  mode: 'online' | 'offline';
  timestamp: Date;
}

interface Conversation {
  id: number;
  characterId: number;
  personaId?: number;
  mode: 'online' | 'offline';
  messages: Message[];
}

interface ChatState {
  conversations: Conversation[];
  activeConversationId: number | null;
  
  // Actions
  setActiveConversation: (id: number) => void;
  addMessage: (conversationId: number, message: Message) => void;
  switchMode: (conversationId: number, mode: 'online' | 'offline') => void;
}

export const useChatStore = create<ChatState>()(
  persist(
    (set) => ({
      conversations: [],
      activeConversationId: null,
      
      setActiveConversation: (id) => set({ activeConversationId: id }),
      
      addMessage: (conversationId, message) => set((state) => ({
        conversations: state.conversations.map(conv =>
          conv.id === conversationId
            ? { ...conv, messages: [...conv.messages, message] }
            : conv
        ),
      })),
      
      switchMode: (conversationId, mode) => set((state) => ({
        conversations: state.conversations.map(conv =>
          conv.id === conversationId ? { ...conv, mode } : conv
        ),
      })),
    }),
    {
      name: 'chat-storage',
    }
  )
);
```

## 11. FAQ & 故障排查

### 11.1 常见问题

**Q: AI响应速度慢怎么办？**
A: 
1. 检查网络连接
2. 减少上下文长度（限制记忆召回数量）
3. 使用更快的模型
4. 启用流式输出

**Q: 记忆检索不准确？**
A:
1. 检查向量嵌入质量
2. 调整检索参数（limit, threshold）
3. 增加记忆的importance权重
4. 定期清理低质量记忆

**Q: 数据库连接失败？**
A:
1. 检查.env配置
2. 确认PostgreSQL服务运行中
3. 检查防火墙设置
